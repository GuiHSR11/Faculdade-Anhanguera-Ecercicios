Calculadora


#include <stdio.h>

int main() {
    float num1, num2;
    char operador;
    float resultado;

   
    printf("Digite o primeiro número real: ");
    scanf("%f", &num1);
    printf("Digite o segundo número real: ");
    scanf("%f", &num2);

   
    printf("Digite o operador (+, -, *, /): ");
    scanf(" %c", &operador);

 
    switch (operador) {
        case '+':
            resultado = num1 + num2;
            break;
        case '-':
            resultado = num1 - num2;
            break;
        case '*':
            resultado = num1 * num2;
            break;
        case '/':
            if (num2 != 0) {
                resultado = num1 / num2;
            } else {
                printf("Erro! Divisão por zero não é permitida.\n");
                return 1; 
            }
            break;
        default:
            printf("Operador inválido!\n");
            return 1; 
    }

    printf("Resultado: %.2f\n", resultado);

    return 0; // Saída do programa sem erro
}
